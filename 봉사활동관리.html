🤝 봉사활동 관리 시스템 — 복사용

시트 2개(봉사자명부 · 봉사활동대장)와 화면 2개(봉사자용 · 담당자용)로 된 시스템입니다. 서버(Code.gs)는 하나이고, 주소 뒤 ?page=admin 여부로 화면이 갈립니다.
설계 원칙
• 화면에선 이름으로 다루지만, 활동 입력 시 ID도 자동 기록됩니다.
• 봉사자 삭제는 비활성(탈퇴) 처리 — 행을 지우지 않아 과거 활동내역이 보존됩니다.
• 활동은 담당자가 입력 → 봉사자가 본인 화면에서 사후 확인합니다.

1. 시트 만들기 (2개)

봉사자명부 — 기존 컬럼 + 상태 추가 (기존 데이터는 '활성'으로). 빈 칸이면 활성으로 간주합니다.

봉사자ID봉사자명연락처소속구분가입일상태

봉사활동대장 — 기존 컬럼 + 확인여부 · 확인시각 추가 (기존 활동은 비워두면 됨).

봉사일자활동유형봉사자ID봉사자명소속구분봉사시간장소비고확인여부확인시각

2. Apps Script 파일 만들기

확장 프로그램 → Apps Script 에서:
① Code.gs 내용을 아래 코드로 교체
② ＋ → HTML → 이름 index → 아래 index 코드 붙여넣기
③ ＋ → HTML → 이름 admin → 아래 admin 코드 붙여넣기
※ HTML 파일 이름은 반드시 index, admin (Code.gs가 이 이름으로 찾습니다)

Code.gs

// ===== 봉사활동 관리 시스템 - Code.gs (서버) =====
// 시트 2개: '봉사자명부', '봉사활동대장'
// 화면 2개: index.html(봉사자용 확인), admin.html(담당자용 관리)

const SHEET_MEMBER = '봉사자명부';   // 봉사자ID, 봉사자명, 연락처, 소속구분, 가입일, 상태
const SHEET_LOG    = '봉사활동대장'; // 봉사일자, 활동유형, 봉사자ID, 봉사자명, 소속구분, 봉사시간, 장소, 비고, 확인여부, 확인시각

// ───────── 화면 라우팅 ─────────
// ?page=admin 이면 담당자 화면, 아니면 봉사자 화면
function doGet(e) {
  const page = (e && e.parameter && e.parameter.page) ? e.parameter.page : 'index';
  const file = (page === 'admin') ? 'admin' : 'index';
  return HtmlService.createHtmlOutputFromFile(file)
    .setTitle(page === 'admin' ? '봉사활동 관리(담당자)' : '봉사활동 확인(봉사자)')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// ───────── 공통 도우미 ─────────
function getSheet_(name) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  if (!sheet) throw new Error('시트를 찾을 수 없습니다: ' + name);
  return sheet;
}

function fmtDate_(v) {
  if (!v) return '';
  if (Object.prototype.toString.call(v) === '[object Date]') {
    return Utilities.formatDate(v, 'GMT+9', 'yyyy-MM-dd');
  }
  return String(v); // 이미 문자열이면 그대로
}

// ───────── 봉사자명부 ─────────
// 전체 봉사자 읽기 (행번호 포함). includeInactive=false면 활성만.
function getMembers(includeInactive) {
  const sheet = getSheet_(SHEET_MEMBER);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];
  const values = sheet.getRange(2, 1, lastRow - 1, 6).getValues();
  const list = values.map(function (r, i) {
    return {
      rowNum: i + 2,
      id: r[0],
      name: r[1],
      phone: r[2],
      group: r[3],
      joinDate: fmtDate_(r[4]),
      status: r[5] || '활성' // 빈 칸이면 활성으로 간주
    };
  });
  return includeInactive ? list : list.filter(function (m) { return m.status !== '탈퇴'; });
}

// 활성 봉사자 이름 목록 (드롭다운용) — id, name만
function getMemberOptions() {
  return getMembers(false).map(function (m) { return { id: m.id, name: m.name, group: m.group }; });
}

// 다음 봉사자ID 자동 생성 (V001, V002 ...)
function nextMemberId_() {
  const members = getMembers(true);
  let max = 0;
  members.forEach(function (m) {
    const n = parseInt(String(m.id).replace(/[^0-9]/g, ''), 10);
    if (!isNaN(n) && n > max) max = n;
  });
  return 'V' + ('000' + (max + 1)).slice(-3);
}

// 봉사자 등록 — 이름/연락처/소속구분 받아서 ID 자동 부여
function addMember(name, phone, group) {
  if (!name) throw new Error('이름을 입력하세요.');
  const sheet = getSheet_(SHEET_MEMBER);
  const id = nextMemberId_();
  const today = Utilities.formatDate(new Date(), 'GMT+9', 'yyyy-MM-dd');
  sheet.appendRow([id, name, phone || '', group || '개인', today, '활성']);
  return getMembers(true);
}

// 봉사자 수정 — 명부 정보만 변경 (활동대장은 안 건드림)
function updateMember(rowNum, name, phone, group) {
  const sheet = getSheet_(SHEET_MEMBER);
  sheet.getRange(rowNum, 2).setValue(name);  // 봉사자명
  sheet.getRange(rowNum, 3).setValue(phone); // 연락처
  sheet.getRange(rowNum, 4).setValue(group); // 소속구분
  return getMembers(true);
}

// 봉사자 비활성(탈퇴) — 행을 지우지 않고 상태만 변경 → 활동내역 보존
function deactivateMember(rowNum) {
  const sheet = getSheet_(SHEET_MEMBER);
  sheet.getRange(rowNum, 6).setValue('탈퇴'); // 상태
  return getMembers(true);
}

// 봉사자 검색 — 이름 또는 ID 부분일치 (활성/탈퇴 모두)
function searchMembers(keyword) {
  const kw = String(keyword || '').trim().toLowerCase();
  const all = getMembers(true);
  if (!kw) return all;
  return all.filter(function (m) {
    return String(m.name).toLowerCase().indexOf(kw) > -1 ||
           String(m.id).toLowerCase().indexOf(kw) > -1;
  });
}

// ───────── 봉사활동대장 ─────────
// 활동 전체 읽기 (행번호 포함)
function getLogs_() {
  const sheet = getSheet_(SHEET_LOG);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];
  const values = sheet.getRange(2, 1, lastRow - 1, 10).getValues();
  return values.map(function (r, i) {
    return {
      rowNum: i + 2,
      date: fmtDate_(r[0]),
      type: r[1],
      id: r[2],
      name: r[3],
      group: r[4],
      hours: Number(r[5]) || 0,
      place: r[6],
      note: r[7],
      confirmed: r[8] || '',        // 확인여부 ('확인' 또는 빈칸)
      confirmTime: r[8] ? fmtDate_(r[9]) + ' ' + (r[9] ? Utilities.formatDate(new Date(r[9]), 'GMT+9', 'HH:mm') : '') : ''
    };
  });
}

// 활동 일괄 입력
// common = {date, type, place}, rows = [{name, id, hours, type?, place?, note?}, ...]
// 줄마다 type/place가 있으면 공통값을 덮어씀
function addActivities(common, rows) {
  if (!rows || !rows.length) throw new Error('입력할 봉사자가 없습니다.');
  const sheet = getSheet_(SHEET_LOG);

  // 이름→소속구분 매핑 (소속을 자동으로 채우기 위해)
  const memberMap = {};
  getMembers(true).forEach(function (m) { memberMap[m.id] = m.group; });

  const toAppend = rows.map(function (row) {
    const type  = row.type  || common.type;
    const place = row.place || common.place;
    const group = memberMap[row.id] || '';
    return [
      common.date,        // 봉사일자
      type,               // 활동유형
      row.id,             // 봉사자ID (자동 기록)
      row.name,           // 봉사자명
      group,              // 소속구분 (명부에서 자동)
      Number(row.hours) || 0, // 봉사시간
      place,              // 장소
      row.note || '',     // 비고
      '',                 // 확인여부 (미확인)
      ''                  // 확인시각
    ];
  });

  // 여러 행을 한 번에 추가 (마지막 행 다음부터)
  const start = sheet.getLastRow() + 1;
  sheet.getRange(start, 1, toAppend.length, 10).setValues(toAppend);
  return { added: toAppend.length };
}

// ───────── 봉사자용 (index) ─────────
// 이름으로 본인 활동내역 조회 + 누적시간
function getMyActivities(name) {
  const nm = String(name || '').trim();
  if (!nm) return { name: nm, total: 0, items: [] };
  const items = getLogs_().filter(function (l) { return String(l.name).trim() === nm; });
  const total = items.reduce(function (s, l) { return s + l.hours; }, 0);
  // 최신 활동이 위로
  items.sort(function (a, b) { return a.date < b.date ? 1 : -1; });
  return { name: nm, total: total, items: items };
}

// 봉사자가 본인 활동을 '확인' 처리
function confirmActivity(rowNum) {
  const sheet = getSheet_(SHEET_LOG);
  sheet.getRange(rowNum, 9).setValue('확인');       // 확인여부
  sheet.getRange(rowNum, 10).setValue(new Date());  // 확인시각
  return true;
}

// ───────── 현황/통계 (admin) ─────────
// 기간(start~end, 'yyyy-MM-dd') 안의 활동을 집계. 비우면 전체.
function getStats(startDate, endDate) {
  let logs = getLogs_();
  if (startDate) logs = logs.filter(function (l) { return l.date >= startDate; });
  if (endDate)   logs = logs.filter(function (l) { return l.date <= endDate; });

  let totalHours = 0;
  const byType = {};
  const byGroup = {};
  const byMonth = {};
  const people = {};

  logs.forEach(function (l) {
    totalHours += l.hours;
    byType[l.type]   = (byType[l.type]   || 0) + l.hours;
    byGroup[l.group] = (byGroup[l.group] || 0) + l.hours;
    const month = String(l.date).slice(0, 7); // yyyy-MM
    byMonth[month] = (byMonth[month] || 0) + l.hours;
    people[l.name] = true;
  });

  return {
    count: logs.length,           // 활동 건수
    people: Object.keys(people).length, // 참여 인원수
    totalHours: totalHours,       // 총 봉사시간
    byType: byType,
    byGroup: byGroup,
    byMonth: byMonth,
    logs: logs                    // 상세 목록(표로 보여줄 용)
  };
}

    function search() {
      const name = document.getElementById('name').value.trim();
      if (!name) { alert('이름을 입력하세요.'); return; }
      const btn = document.getElementById('searchBtn');
      btn.disabled = true; btn.textContent = '조회 중...';
      document.getElementById('status').textContent = '조회 중...';
      google.script.run
        .withSuccessHandler(function (res) {
          showResult(res);
          btn.disabled = false; btn.textContent = '내 활동 조회';
        })
        .withFailureHandler(function (err) {
          document.getElementById('status').textContent = '조회 실패: ' + err.message;
          btn.disabled = false; btn.textContent = '내 활동 조회';
        })
        .getMyActivities(name);
    }

    function showResult(res) {
      const status = document.getElementById('status');
      const summary = document.getElementById('summary');
      const list = document.getElementById('list');

      if (!res.items || res.items.length === 0) {
        status.textContent = '"' + res.name + '" 님의 활동 기록이 없습니다.';
        summary.innerHTML = '';
        list.innerHTML = '';
        return;
      }

      status.textContent = res.name + ' 님의 봉사활동';
      summary.innerHTML = '누적 봉사시간 <b>' + res.total + '</b> 시간 · 총 ' + res.items.length + '건';

      list.innerHTML = res.items.map(function (it) {
        const right = (it.confirmed === '확인')
          ? '<span class="confirmed">✓ 확인완료<br>' + it.confirmTime + '</span>'
          : '<button class="confirm-btn" onclick="confirmIt(' + it.rowNum + ', this)">확인</button>';
        return '<div class="item">' +
                 '<div class="info">' +
                   '<b>' + escapeHtml(it.date) + '</b> · ' + escapeHtml(it.type) + ' · ' + it.hours + '시간' +
                   '<div class="meta">' + escapeHtml(it.place || '') + (it.note ? ' · ' + escapeHtml(it.note) : '') + '</div>' +
                 '</div>' +
                 right +
               '</div>';
      }).join('');
    }

    function confirmIt(rowNum, btn) {
      btn.disabled = true; btn.textContent = '처리 중...';
      google.script.run
        .withSuccessHandler(function () { search(); }) // 확인 후 목록 새로고침
        .withFailureHandler(function (err) {
          alert('확인 실패: ' + err.message);
          btn.disabled = false; btn.textContent = '확인';
        })
        .confirmActivity(rowNum);
    }

    function escapeHtml(t) {
      const d = document.createElement('div'); d.textContent = t == null ? '' : t; return d.innerHTML;
    }

    function loadMembers() {
      google.script.run
        .withSuccessHandler(function (list) {
          MEMBERS = list;

    function renderMembers() {
      const kw = document.getElementById('m_search').value.trim().toLowerCase();
      const rows = MEMBERS.filter(function (m) {
        if (!kw) return true;
        return String(m.name).toLowerCase().indexOf(kw) > -1 ||
               String(m.id).toLowerCase().indexOf(kw) > -1;
      });
      const tbody = document.querySelector('#memberTable tbody');
      if (rows.length === 0) { tbody.innerHTML = '<tr><td class="muted">결과 없음</td></tr>'; return; }

          : '<button class="sub" onclick="editMember(' + m.rowNum + ')">수정</button> ' +

    function showStats(s) {
      function distTable(title, obj) {
        const keys = Object.keys(obj).sort();
        if (keys.length === 0) return '';
        let h = '<div class="box"><b>' + title + '</b><table><tr><th>구분</th><th>봉사시간</th></tr>';
        h += keys.map(function (k){ return '<tr><td>' + esc(k) + '</td><td>' + obj[k] + '</td></tr>'; }).join('');
        return h + '</table></div>';
      }
      let html = '<div class="stat-cards">' +
        card(s.totalHours, '총 봉사시간') +
        card(s.people, '참여 인원') +
        card(s.count, '활동 건수') +
        '</div>';
      html += distTable('월별', s.byMonth);
      html += distTable('활동유형별', s.byType);
      html += distTable('소속구분별', s.byGroup);

      // 상세 목록
      html += '<div class="box"><b>상세 내역 (' + s.logs.length + '건)</b><table>' +
        '<tr><th>일자</th><th>유형</th><th>이름</th><th>소속</th><th>시간</th><th>장소</th><th>확인</th></tr>';
      const sorted = s.logs.slice().sort(function(a,b){ return a.date<b.date?1:-1; });
      html += sorted.map(function (l) {
        return '<tr><td>' + esc(l.date) + '</td><td>' + esc(l.type) + '</td><td>' + esc(l.name) + '</td>' +
               '<td>' + esc(l.group) + '</td><td>' + l.hours + '</td><td>' + esc(l.place) + '</td>' +
               '<td>' + (l.confirmed === '확인' ? '✓' : '<span class="muted">-</span>') + '</td></tr>';
      }).join('');
      html += '</table></div>';
      document.getElementById('statsResult').innerHTML = html;
    }

3. 배포 & 두 화면 주소

배포 → 새 배포 → 웹 앱 → 실행: 나 / 액세스: 모든 사용자 → 배포 → 권한 허용

나온 웹앱 주소가 봉사자용입니다.
그 주소 맨 뒤에 ?page=admin 을 붙이면 담당자용 화면이 열립니다.
예: .../exec → 봉사자용  |  .../exec?page=admin → 담당자용

⚠️ 코드 수정 후에는 배포 관리 → 편집(연필) → 새 버전 → 배포.
