🤖 방명록 LV4 — AI 환영 코멘트

방명록에 글이 올라오면 AI가 자동으로 환영 코멘트를 달아주는 버전입니다. 모델은 gemini-2.5-flash-lite (구글 AI 스튜디오), API 키는 스크립트 속성에 저장합니다.
이번 버전에 추가된 것
① 시트에 E열(환영코멘트) 추가
② 글 저장 시 Gemma API 호출로 환영 코멘트 생성 → E열에 저장 (Code.gs)
③ 각 글 아래 🤖 AI 코멘트 표시 (index.html)
④ AI 호출이 실패해도 방명록 글은 정상 저장 (데이터를 잃지 않음)
🔐 이번 단계의 핵심 — 안전한 AI 사용
API 키를 코드나 화면(index.html)에 절대 적지 않습니다. 키는 서버(스크립트 속성)에만 두고, AI 호출도 서버(Code.gs)에서만 합니다. 그래서 교육생이 F12로 들여다봐도 키가 보이지 않습니다.
또한 AI에 보내는 것은 방명록의 이름·메시지입니다. "무엇을 AI에 보내는가"를 늘 의식하세요.

1. API 키 발급 (구글 AI 스튜디오)

① aistudio.google.com 접속 → 로그인
② 왼쪽 Get API key → Create API key
③ 생성된 키를 복사 (이 키는 비밀번호처럼 다룰 것)

2. 키를 스크립트 속성에 저장

Apps Script 에디터 → 왼쪽 프로젝트 설정(톱니바퀴 ⚙️) → 맨 아래 스크립트 속성
→ 속성 추가 → 이름: GEMINI_API_KEY / 값: (복사한 키 붙여넣기) → 저장

※ 이름은 반드시 GEMINI_API_KEY (Code.gs가 이 이름으로 키를 찾습니다.)

3. 구글시트 제목행 (1행)

시트 탭 이름 방명록. 기존 A~D에 E열(환영코멘트)이 추가됐습니다.

4. Code.gs (서버 코드)

AI 호출 + 키 읽기가 들어간 버전. testWelcome 함수를 에디터에서 실행하면 키·모델이 잘 되는지 미리 확인할 수 있습니다.

// ===== 구글시트 방명록 - Code.gs (서버 코드) =====
// 구글시트를 데이터베이스로 사용하는 방명록 웹앱

// 시트(탭) 이름 — 구글시트 아래쪽 탭 이름과 반드시 똑같이!
const SHEET_NAME = '방명록';

// 사용할 AI 모델 (구글 AI 스튜디오 / Generative Language API)
const AI_MODEL = 'gemini-2.5-flash-lite';

// 스크립트 속성에서 API 키를 꺼냄 (코드·화면에 키를 직접 안 적음!)
function getApiKey_() {
  const key = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
  if (!key) {
    throw new Error('스크립트 속성에 GEMINI_API_KEY가 없습니다. 프로젝트 설정 → 스크립트 속성에서 추가하세요.');
  }
  return key;
}

// [AI] 방명록 글에 다는 환영 코멘트 생성 (실패해도 빈 문자열 반환 → 방명록은 살림)
function generateWelcome_(name, message) {
  try {
    const url = 'https://generativelanguage.googleapis.com/v1beta/models/'
      + AI_MODEL + ':generateContent?key=' + getApiKey_();

    // 경량 모델이 '번역·요약'으로 새지 않도록 예시(few-shot)로 패턴을 박아줌
    const prompt =
      '너는 방명록에 한국어 환영 인사를 다는 역할이야. 아래 예시처럼, 마지막 "환영 인사:" 뒤에 ' +
      '한국어 한 문장만 써. 번역·설명·영어 금지. 이모지는 한 개까지.\n\n' +
      '메시지: 처음 와봤어요\n' +
      '환영 인사: 처음 오셨군요, 진심으로 환영합니다! 😊\n\n' +
      '메시지: 잘 보고 갑니다\n' +
      '환영 인사: 들러주셔서 정말 고마워요, 또 놀러 오세요!\n\n' +
      '메시지: 화이팅입니다\n' +
      '환영 인사: 따뜻한 응원 감사해요, 덕분에 힘이 나네요!\n\n' +
      '메시지: ' + message + '\n' +
      '환영 인사:';

    const payload = {
      contents: [{ parts: [{ text: prompt }] }],
      generationConfig: { temperature: 0.9, maxOutputTokens: 200, stopSequences: ['메시지:'] }
    };

    const res = UrlFetchApp.fetch(url, {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify(payload),
      muteHttpExceptions: true // 에러여도 예외 안 던지고 응답을 받음
    });

    if (res.getResponseCode() !== 200) {
      console.error('AI 호출 실패: ' + res.getResponseCode() + ' ' + res.getContentText());
      return ''; // 실패 시 코멘트 없이 진행
    }

    const data = JSON.parse(res.getContentText());
    let text = data.candidates[0].content.parts[0].text.trim();
    // 모델이 앞에 '환영 인사:' 같은 라벨이나 따옴표·별표를 붙이면 제거
    text = text.replace(/^환영\s*인사\s*[:：]?\s*/i, '')
               .replace(/^[*"'\s]+|[*"'\s]+$/g, '')
               .split('\n')[0]; // 첫 줄만 사용
    return text;
  } catch (e) {
    console.error('AI 코멘트 생성 중 오류: ' + e.message);
    return ''; // 어떤 오류든 방명록 저장은 막지 않음
  }
}

// (선택) 에디터에서 직접 실행해 키·모델이 잘 작동하는지 확인하는 테스트
function testWelcome() {
  Logger.log(generateWelcome_('홍길동', '안녕하세요, 잘 부탁드립니다!'));
}

// [권한 점검] 외부 호출 등 권한이 충분한지 확인하고, 부족하면 재승인 URL을 알려줌
// 사용법: 에디터에서 이 함수를 실행한 뒤 실행 로그를 확인 → 안내대로 처리
function checkAuth() {
  const info = ScriptApp.getAuthorizationInfo(ScriptApp.AuthMode.FULL);
  if (info.getAuthorizationStatus() === ScriptApp.AuthorizationStatus.REQUIRED) {
    Logger.log('⚠️ 권한이 부족합니다. 아래 주소를 열어 다시 승인하세요:');
    Logger.log(info.getAuthorizationUrl());
  } else {
    Logger.log('✅ 권한 정상입니다. 추가 승인이 필요 없습니다.');
  }
}

// 웹앱 주소로 접속하면 index.html 화면을 보여줌
function doGet() {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle('우리 반 방명록')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// [저장] 방명록에 글 한 줄 추가하기 (+ AI 환영 코멘트)
function addEntry(name, message) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const welcome = generateWelcome_(name, message); // AI가 환영 코멘트 생성
  sheet.appendRow([new Date(), name, message, 0, welcome]); // [시간,이름,메시지,좋아요,환영코멘트]
  return getEntries(); // 저장 후 최신 목록을 돌려줌
}

// [읽기] 방명록 글 전부 가져오기 (최신 글이 위로)
function getEntries() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return []; // 헤더(1행)만 있으면 빈 목록

  const values = sheet.getRange(2, 1, lastRow - 1, 5).getValues(); // A~E열(5칸)
  const entries = values.map(function (row, i) {
    return {
      rowNum: i + 2, // 실제 시트 행 번호 (좋아요 누를 때 이 줄을 찾음)
      time: Utilities.formatDate(new Date(row[0]), 'GMT+9', 'yyyy-MM-dd HH:mm'),
      name: row[1],
      message: row[2],
      likes: row[3] || 0, // 좋아요 수 (빈 칸이면 0)
      welcome: row[4] || '' // AI 환영 코멘트 (없으면 빈 문자열)
    };
  });
  return entries.reverse(); // 최신 글이 맨 위로
}

// [좋아요] 특정 줄의 좋아요를 1 올리기 (동시 클릭 안전 - 락 적용)
function likeEntry(rowNum) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000); // 한 번에 한 명만 들어오게, 최대 10초 대기
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
    const cell = sheet.getRange(rowNum, 4); // 그 줄의 D열(좋아요)
    const current = cell.getValue() || 0;
    cell.setValue(current + 1);
    SpreadsheetApp.flush(); // 시트에 즉시 반영
    return getEntries(); // 올린 뒤 최신 목록을 돌려줌
  } finally {
    lock.releaseLock(); // 끝나면 반드시 락 풀기
  }
}

<body>
  <h1>📖 우리 반 방명록</h1>
  <button id="pollBtn" class="poll-btn" onclick="togglePolling()">자동 새로고침: 켜짐</button>

  <div class="form-box">
    <input type="text" id="name" placeholder="이름" maxlength="20">
    <textarea id="message" placeholder="메시지를 남겨주세요" rows="3" maxlength="200"></textarea>
    <button id="submitBtn" onclick="submitEntry()">남기기</button>
  </div>

  <p id="status">불러오는 중...</p>
  <div id="list"></div>

  <script>
    var pollTimer = null; // 자동 새로고침 타이머 (켜져 있으면 값이 들어감)

    // 목록을 서버에서 받아와 화면을 갱신 (재사용)
    function loadEntries() {
      google.script.run
        .withSuccessHandler(showEntries)
        .withFailureHandler(function (err) {
          document.getElementById('status').textContent = '불러오기 실패: ' + err.message;
        })
        .getEntries();
    }

    // 자동 새로고침(폴링) 켜기/끄기
    function togglePolling() {
      const btn = document.getElementById('pollBtn');
      if (pollTimer) {
        clearInterval(pollTimer); // 끄기
        pollTimer = null;
        btn.textContent = '자동 새로고침: 꺼짐';
      } else {
        pollTimer = setInterval(loadEntries, 5000); // 켜기 (5초마다)
        btn.textContent = '자동 새로고침: 켜짐';
      }
    }

    // 페이지가 열리면 한 번 불러오고, 자동 새로고침은 켠 상태로 시작
    window.onload = function () {
      loadEntries();
      pollTimer = setInterval(loadEntries, 5000); // 5초마다 → 남이 쓴 글/좋아요도 자동 반영
    };

    // [남기기] 버튼을 누르면 실행
    function submitEntry() {
      const name = document.getElementById('name').value.trim();
      const message = document.getElementById('message').value.trim();
      if (!name || !message) {
        alert('이름과 메시지를 모두 입력해주세요.');
        return;
      }

      const btn = document.getElementById('submitBtn');
      btn.disabled = true;
      btn.textContent = '저장 중...';

      // 서버(Code.gs)의 addEntry 함수를 호출 → 끝나면 목록 새로고침
      google.script.run
        .withSuccessHandler(function (entries) {
          document.getElementById('message').value = '';
          showEntries(entries);
          btn.disabled = false;
          btn.textContent = '남기기';
        })
        .withFailureHandler(function (err) {
          alert('저장 실패: ' + err.message);
          btn.disabled = false;
          btn.textContent = '남기기';
        })
        .addEntry(name, message);
    }

    // 받아온 목록을 화면에 그리기
    function showEntries(entries) {
      const status = document.getElementById('status');
      const list = document.getElementById('list');

      if (!entries || entries.length === 0) {
        status.textContent = '아직 글이 없어요. 첫 글을 남겨보세요!';
        list.innerHTML = '';
        return;
      }

      status.textContent = '총 ' + entries.length + '개의 글';
      list.innerHTML = entries.map(function (e) {
        // AI 환영 코멘트가 있으면 메시지 아래에 표시
        var welcomeHtml = e.welcome
          ? '<div class="welcome">🤖 ' + escapeHtml(e.welcome) + '</div>'
          : '';
        return '<div class="entry">' +
                 '<div class="meta">' + escapeHtml(e.name) + ' · ' + e.time + '</div>' +
                 '<div class="msg">' + escapeHtml(e.message) + '</div>' +
                 welcomeHtml +
                 '<button class="like-btn" onclick="likePost(' + e.rowNum + ')">❤️ ' + e.likes + '</button>' +
               '</div>';
      }).join('');
    }

    // 하트 버튼을 누르면 그 글의 좋아요를 1 올림
    function likePost(rowNum) {
      google.script.run
        .withSuccessHandler(showEntries)
        .withFailureHandler(function (err) { alert('좋아요 실패: ' + err.message); })
        .likeEntry(rowNum);
    }

    // 입력값의 특수문자를 안전하게 표시 (화면 깨짐·악용 방지)
    function escapeHtml(text) {
      const div = document.createElement('div');
      div.textContent = text;
      return div.innerHTML;
    }
  </script>
</body>
</html>

6. 권한 승인 + 작동 테스트 (코드 다 붙여넣은 뒤)

AI 기능은 외부 인터넷 호출 권한이 필요합니다. 새로 생긴 권한이라 다시 승인해야 하는데, 권한 창이 안 뜨는 경우가 많아 아래 함수로 처리합니다.
① 에디터 상단 함수 드롭다운에서 checkAuth 선택 → ▶실행
② 실행 로그(보기 → 실행 로그) 확인
   • "⚠️ 권한이 부족합니다" + 링크가 보이면 → 그 링크를 열어 승인 (고급 → 허용)
   • "✅ 권한 정상입니다" 면 → 그대로 진행
③ 함수 드롭다운에서 testWelcome 선택 → ▶실행 → 로그에 한국어 환영 인사 한 문장이 찍히면 성공

※ 로그가 영어 분석·메타설명으로 나오면 모델이 안 맞는 것 → Code.gs 상단 AI_MODEL을 gemini-2.5-flash-lite로 (현재 적용됨).

⚠️ 코드 수정 후에는 반드시 배포 → 배포 관리 → 편집(연필) → 버전: 새 버전 → 배포.
