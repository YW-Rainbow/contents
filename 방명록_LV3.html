📖 방명록 LV3

2단계 방명록에 두 기능을 얹은 버전입니다. 코드를 붙여넣은 뒤 반드시 재배포(배포 → 배포 관리 → 편집 → 새 버전)해야 적용됩니다.
이번 버전의 핵심 2가지
① 자동 새로고침 토글 — 화면 위 버튼으로 5초 폴링을 켜고/끌 수 있음. 끄면 호출이 멈춰 할당량을 아낌 (index.html)
② LockService — 여러 명이 동시에 좋아요를 눌러도 숫자가 새지 않게 함 (Code.gs의 likeEntry)
※ 참고: GAS는 WebSocket/SSE 같은 진짜 실시간(push)을 지원하지 않습니다. 폴링이 GAS 안에서의 최선이며, 진짜 실시간은 Firebase/Supabase로 갑니다.

1. 구글시트 제목행 (1행)

시트 탭 이름 방명록, A1~D1에 아래 입력.

시간이름메시지좋아요
시간	이름	메시지	좋아요

2. Code.gs (서버 코드)

likeEntry에 LockService가 들어간 버전입니다.

// ===== 구글시트 방명록 - Code.gs (서버 코드) =====
// 구글시트를 데이터베이스로 사용하는 방명록 웹앱

// 시트(탭) 이름 — 구글시트 아래쪽 탭 이름과 반드시 똑같이!
const SHEET_NAME = '방명록';

// 웹앱 주소로 접속하면 index.html 화면을 보여줌
function doGet() {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle('우리 반 방명록')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// [저장] 방명록에 글 한 줄 추가하기
function addEntry(name, message) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  sheet.appendRow([new Date(), name, message, 0]); // [시간, 이름, 메시지, 좋아요(0)]
  return getEntries(); // 저장 후 최신 목록을 돌려줌
}

// [읽기] 방명록 글 전부 가져오기 (최신 글이 위로)
function getEntries() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return []; // 헤더(1행)만 있으면 빈 목록

  const values = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); // A~D열(4칸)
  const entries = values.map(function (row, i) {
    return {
      rowNum: i + 2, // 실제 시트 행 번호 (좋아요 누를 때 이 줄을 찾음)
      time: Utilities.formatDate(new Date(row[0]), 'GMT+9', 'yyyy-MM-dd HH:mm'),
      name: row[1],
      message: row[2],
      likes: row[3] || 0 // 좋아요 수 (빈 칸이면 0)
    };
  });
  return entries.reverse(); // 최신 글이 맨 위로
}

// [좋아요] 특정 줄의 좋아요를 1 올리기 (동시 클릭 안전 - 락 적용)
function likeEntry(rowNum) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000); // 한 번에 한 명만 들어오게, 최대 10초 대기
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
    const cell = sheet.getRange(rowNum, 4); // 그 줄의 D열(좋아요)
    const current = cell.getValue() || 0;
    cell.setValue(current + 1);
    SpreadsheetApp.flush(); // 시트에 즉시 반영
    return getEntries(); // 올린 뒤 최신 목록을 돌려줌
  } finally {
    lock.releaseLock(); // 끝나면 반드시 락 풀기
  }
}

3. index.html (화면)

자동 새로고침 토글 버튼 + 5초 폴링이 들어간 버전입니다.

<!DOCTYPE html>
<html>
<head>
  <base target="_top">
  <meta charset="utf-8">
  <style>
    body {
      font-family: -apple-system, 'Apple SD Gothic Neo', 'Malgun Gothic', sans-serif;
      max-width: 600px; margin: 0 auto; padding: 20px; background: #f5f5f7; color: #222;
    }
    h1 { font-size: 22px; }
    .form-box { background: #fff; padding: 16px; border-radius: 12px; margin-bottom: 20px; }
    input, textarea {
      width: 100%; box-sizing: border-box; padding: 10px; margin-bottom: 8px;
      border: 1px solid #ddd; border-radius: 8px; font-size: 14px; font-family: inherit;
    }
    textarea { resize: vertical; max-width: 100%; }
    button {
      background: #2f9e6e; color: #fff; border: none; padding: 10px 20px;
      border-radius: 8px; font-size: 14px; cursor: pointer;
    }
    button:disabled { background: #aaa; cursor: default; }
    .entry { background: #fff; padding: 12px 16px; border-radius: 12px; margin-bottom: 10px; }
    .entry .meta { font-size: 12px; color: #888; margin-bottom: 4px; }
    .entry .msg { font-size: 15px; white-space: pre-wrap; }
    .like-btn { margin-top: 8px; background: #fff; border: 1px solid #e0a; color: #c07;
                border-radius: 20px; padding: 4px 12px; font-size: 13px; cursor: pointer; }
    .like-btn:active { background: #fce; }
    #status { color: #888; font-size: 13px; }
    .poll-btn { margin: 0 0 12px; background: #eef; border: 1px solid #ccd; color: #335;
                border-radius: 6px; padding: 3px 10px; font-size: 12px; cursor: pointer; }
  </style>
</head>
<body>
  <h1>📖 우리 반 방명록</h1>
  <button id="pollBtn" class="poll-btn" onclick="togglePolling()">자동 새로고침: 켜짐</button>

  <div class="form-box">
    <input type="text" id="name" placeholder="이름" maxlength="20">
    <textarea id="message" placeholder="메시지를 남겨주세요" rows="3" maxlength="200"></textarea>
    <button id="submitBtn" onclick="submitEntry()">남기기</button>
  </div>

  <p id="status">불러오는 중...</p>
  <div id="list"></div>

  <script>
    var pollTimer = null; // 자동 새로고침 타이머 (켜져 있으면 값이 들어감)

    // 목록을 서버에서 받아와 화면을 갱신 (재사용)
    function loadEntries() {
      google.script.run
        .withSuccessHandler(showEntries)
        .withFailureHandler(function (err) {
          document.getElementById('status').textContent = '불러오기 실패: ' + err.message;
        })
        .getEntries();
    }

    // 자동 새로고침(폴링) 켜기/끄기
    function togglePolling() {
      const btn = document.getElementById('pollBtn');
      if (pollTimer) {
        clearInterval(pollTimer); // 끄기
        pollTimer = null;
        btn.textContent = '자동 새로고침: 꺼짐';
      } else {
        pollTimer = setInterval(loadEntries, 5000); // 켜기 (5초마다)
        btn.textContent = '자동 새로고침: 켜짐';
      }
    }

    // 페이지가 열리면 한 번 불러오고, 자동 새로고침은 켠 상태로 시작
    window.onload = function () {
      loadEntries();
      pollTimer = setInterval(loadEntries, 5000); // 5초마다 → 남이 쓴 글/좋아요도 자동 반영
    };

    // [남기기] 버튼을 누르면 실행
    function submitEntry() {
      const name = document.getElementById('name').value.trim();
      const message = document.getElementById('message').value.trim();
      if (!name || !message) {
        alert('이름과 메시지를 모두 입력해주세요.');
        return;
      }

      const btn = document.getElementById('submitBtn');
      btn.disabled = true;
      btn.textContent = '저장 중...';

      // 서버(Code.gs)의 addEntry 함수를 호출 → 끝나면 목록 새로고침
      google.script.run
        .withSuccessHandler(function (entries) {
          document.getElementById('message').value = '';
          showEntries(entries);
          btn.disabled = false;
          btn.textContent = '남기기';
        })
        .withFailureHandler(function (err) {
          alert('저장 실패: ' + err.message);
          btn.disabled = false;
          btn.textContent = '남기기';
        })
        .addEntry(name, message);
    }

    // 받아온 목록을 화면에 그리기
    function showEntries(entries) {
      const status = document.getElementById('status');
      const list = document.getElementById('list');

      if (!entries || entries.length === 0) {
        status.textContent = '아직 글이 없어요. 첫 글을 남겨보세요!';
        list.innerHTML = '';
        return;
      }

      status.textContent = '총 ' + entries.length + '개의 글';
      list.innerHTML = entries.map(function (e) {
        return '<div class="entry">' +
                 '<div class="meta">' + escapeHtml(e.name) + ' · ' + e.time + '</div>' +
                 '<div class="msg">' + escapeHtml(e.message) + '</div>' +
                 '<button class="like-btn" onclick="likePost(' + e.rowNum + ')">❤️ ' + e.likes + '</button>' +
               '</div>';
      }).join('');
    }

    // 하트 버튼을 누르면 그 글의 좋아요를 1 올림
    function likePost(rowNum) {
      google.script.run
        .withSuccessHandler(showEntries)
        .withFailureHandler(function (err) { alert('좋아요 실패: ' + err.message); })
        .likeEntry(rowNum);
    }

    // 입력값의 특수문자를 안전하게 표시 (화면 깨짐·악용 방지)
    function escapeHtml(text) {
      const div = document.createElement('div');
      div.textContent = text;
      return div.innerHTML;
    }
  </script>
</body>
</html>

⚠️ 수정 후에는 반드시 배포 → 배포 관리 → 편집(연필) → 버전: 새 버전 → 배포.
