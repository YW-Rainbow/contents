📖 방명록 만들기 LV1

구글시트를 데이터베이스로, GAS를 서버로 쓰는 방명록입니다. 순서: ① 구글시트 제목행 입력 → ② Code.gs 붙여넣기 → ③ index.html 붙여넣기 → ④ 배포. 각 코드 박스 오른쪽 위 복사 버튼을 누르면 전체가 복사됩니다.

1. 구글시트 제목행 (1행에 입력)

새 구글시트의 시트 탭 이름을 방명록으로 바꾸고, A1·B1·C1 칸에 아래 세 가지를 입력하세요. (복사 버튼을 누른 뒤 A1 칸에 붙여넣으면 세 칸에 한 번에 들어갑니다.)

시간이름메시지
시간	이름	메시지

2. Code.gs (서버 코드)

확장 프로그램 → Apps Script → 기본 Code.gs 내용을 모두 지우고 아래를 붙여넣기.

// ===== 구글시트 방명록 - Code.gs (서버 코드) =====
// 구글시트를 데이터베이스로 사용하는 방명록 웹앱

// 시트(탭) 이름 — 구글시트 아래쪽 탭 이름과 반드시 똑같이!
const SHEET_NAME = '방명록';

// 웹앱 주소로 접속하면 index.html 화면을 보여줌
function doGet() {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle('우리 반 방명록')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// [저장] 방명록에 글 한 줄 추가하기
function addEntry(name, message) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  sheet.appendRow([new Date(), name, message]); // [시간, 이름, 메시지]
  return getEntries(); // 저장 후 최신 목록을 돌려줌
}

// [읽기] 방명록 글 전부 가져오기 (최신 글이 위로)
function getEntries() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return []; // 헤더(1행)만 있으면 빈 목록

  const values = sheet.getRange(2, 1, lastRow - 1, 3).getValues();
  const entries = values.map(function (row) {
    return {
      time: Utilities.formatDate(new Date(row[0]), 'GMT+9', 'yyyy-MM-dd HH:mm'),
      name: row[1],
      message: row[2]
    };
  });
  return entries.reverse(); // 최신 글이 맨 위로
}

3. index.html (화면)

Apps Script 왼쪽 ＋ → HTML → 파일 이름을 index로 만들고, 그 안에 아래를 붙여넣기.

<!DOCTYPE html>
<html>
<head>
  <base target="_top">
  <meta charset="utf-8">
  <style>
    body {
      font-family: -apple-system, 'Apple SD Gothic Neo', 'Malgun Gothic', sans-serif;
      max-width: 600px; margin: 0 auto; padding: 20px; background: #f5f5f7; color: #222;
    }
    h1 { font-size: 22px; }
    .form-box { background: #fff; padding: 16px; border-radius: 12px; margin-bottom: 20px; }
    input, textarea {
      width: 100%; box-sizing: border-box; padding: 10px; margin-bottom: 8px;
      border: 1px solid #ddd; border-radius: 8px; font-size: 14px; font-family: inherit;
    }
    textarea { resize: vertical; max-width: 100%; }
    button {
      background: #2f9e6e; color: #fff; border: none; padding: 10px 20px;
      border-radius: 8px; font-size: 14px; cursor: pointer;
    }
    button:disabled { background: #aaa; cursor: default; }
    .entry { background: #fff; padding: 12px 16px; border-radius: 12px; margin-bottom: 10px; }
    .entry .meta { font-size: 12px; color: #888; margin-bottom: 4px; }
    .entry .msg { font-size: 15px; white-space: pre-wrap; }
    #status { color: #888; font-size: 13px; }
  </style>
</head>
<body>
  <h1>📖 우리 반 방명록</h1>

  <div class="form-box">
    <input type="text" id="name" placeholder="이름" maxlength="20">
    <textarea id="message" placeholder="메시지를 남겨주세요" rows="3" maxlength="200"></textarea>
    <button id="submitBtn" onclick="submitEntry()">남기기</button>
  </div>

  <p id="status">불러오는 중...</p>
  <div id="list"></div>

  <script>
    // 페이지가 열리면 목록부터 불러오기
    window.onload = function () {
      google.script.run
        .withSuccessHandler(showEntries)
        .withFailureHandler(function (err) {
          document.getElementById('status').textContent = '불러오기 실패: ' + err.message;
        })
        .getEntries();
    };

    // [남기기] 버튼을 누르면 실행
    function submitEntry() {
      const name = document.getElementById('name').value.trim();
      const message = document.getElementById('message').value.trim();
      if (!name || !message) {
        alert('이름과 메시지를 모두 입력해주세요.');
        return;
      }

      const btn = document.getElementById('submitBtn');
      btn.disabled = true;
      btn.textContent = '저장 중...';

      // 서버(Code.gs)의 addEntry 함수를 호출 → 끝나면 목록 새로고침
      google.script.run
        .withSuccessHandler(function (entries) {
          document.getElementById('message').value = '';
          showEntries(entries);
          btn.disabled = false;
          btn.textContent = '남기기';
        })
        .withFailureHandler(function (err) {
          alert('저장 실패: ' + err.message);
          btn.disabled = false;
          btn.textContent = '남기기';
        })
        .addEntry(name, message);
    }

    // 받아온 목록을 화면에 그리기
    function showEntries(entries) {
      const status = document.getElementById('status');
      const list = document.getElementById('list');

      if (!entries || entries.length === 0) {
        status.textContent = '아직 글이 없어요. 첫 글을 남겨보세요!';
        list.innerHTML = '';
        return;
      }

      status.textContent = '총 ' + entries.length + '개의 글';
      list.innerHTML = entries.map(function (e) {
        return '<div class="entry">' +
                 '<div class="meta">' + escapeHtml(e.name) + ' · ' + e.time + '</div>' +
                 '<div class="msg">' + escapeHtml(e.message) + '</div>' +
               '</div>';
      }).join('');
    }

    // 입력값의 특수문자를 안전하게 표시 (화면 깨짐·악용 방지)
    function escapeHtml(text) {
      const div = document.createElement('div');
      div.textContent = text;
      return div.innerHTML;
    }
  </script>
</body>
</html>
